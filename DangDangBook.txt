/*订单表*/;

CREATE TABLE `myorder` (
  `orderID` int(11) NOT NULL auto_increment,
  `email` char(30) DEFAULT NULL,
  `custID` int(11) DEFAULT NULL,
  `totalPay` decimal(10,2) DEFAULT NULL,
  `orderStatus` varchar(20) DEFAULT NULL,
  `packageID` varchar(50) DEFAULT NULL,
  `createTime` varchar(100) DEFAULT NULL,
PRIMARY KEY(orderID)

) ENGINE=MyISAM DEFAULT CHARSET=utf8;


/*订单商品列表*/;

CREATE TABLE `shoplist` (
  `SLID` int(11) NOT NULL auto_increment,
  `orderID` int(11) DEFAULT NULL,
  `bookID` char(30) DEFAULT NULL,
  `num` int(11) DEFAULT NULL,
`ifComment` int(11) DEFAULT NULL,
PRIMARY KEY(SLID)

) ENGINE=InnoDB DEFAULT CHARSET=utf8;

/*评论表*/;
CREATE TABLE comment(
	commentID  int  auto_increment,
	email char(30),
	bookID char(15),
	content varchar(200),
	sendTime varchar(100),
	reply varchar(200),
	score int,
       	PRIMARY KEY(commentID)
)ENGINE=MyISAM DEFAULT CHARSET=utf8;

/*收藏表*/
CREATE TABLE `collection` (
  `collectID` int(11) NOT NULL AUTO_INCREMENT,
  `email` char(30) DEFAULT NULL,
  `bookID` char(15) DEFAULT NULL,
  PRIMARY KEY (`collectID`)
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8;

/*购物车视图*/;
create view showcart 
as
select cartID,cart.bookID,bPicture,email,cart.num,bName,bAuthor,bPublisher,bPubTime,bPriceB,bPriceS
from cart,book
where book.bookID=cart.bookID;

create view showorder
as
select myorder.orderID,myorder.email,createTime,custName,custMobile,custAddress,totalPay,packageID,orderStatus from myorder, customerCopy 
where myorder.custID= customerCopy.custID  order by createTime desc,orderID desc;

create view showshoplist 
As 
Select orderID,SLID,num,a.bookID,bName,bAuthor,bPublisher,bPubTime,bPicture,bPriceB,bPriceS,ifComment from shoplist a,book b where  a.bookID=b.bookID;

create view showcollection
As 
select `book`.`bName` AS `bName`,`book`.`bPriceS` AS `bPriceS`,`book`.`bPicture` AS `bPicture`,`collection`.`email` AS `email`,`book`.`bookID` AS `bookID` from (`book` join `collection`) where (`book`.`bookID` = `collection`.`bookID`);
